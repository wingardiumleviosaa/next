[Golang] 使用 winrm 下多行指令操作遠端機器

Publish on: 2023-01-28 Classify at: Programming/Golang
Words: 206 Read:≈ 1mins

TL; DR

使用 masterzen/winrm 套件透過 winrm protocal 遠端連線至指定機器並下 powershell 指令操作。

程式碼 

 1// 直接寫成一個 function~
 2func WinExec() {
 3	params := winrm.DefaultParameters
 4
 5	endpoint := winrm.NewEndpoint("10.37.36.79", 5985, false, false, nil, nil, nil, 0)
 6	client, err := winrm.NewClientWithParameters(endpoint, "Administrator", "Wistron@2017", params)
 7	if err != nil {
 8		panic(err)
 9	}
10
11	ctx, cancel := context.WithCancel(context.Background())
12	defer cancel()
13
14	shell, err := client.CreateShell()
15	if err != nil {
16		panic(err)
17	}
18	defer shell.Close()
19
20	// 可以在這邊下多行指令
21	script := fmt.Sprintf(`
22	ipconfig
23	`)
24
25	cmd, err := shell.ExecuteWithContext(ctx, winrm.Powershell(script))
26	// _, err = client.RunWithContext(ctx, "echo username", os.Stdout, os.Stderr)
27
28	if err != nil {
29		panic(err)
30	}
31
32	defer cmd.Close()
33
34	var wg sync.WaitGroup
35	copyFunc := func(w io.Writer, r io.Reader) {
36		defer wg.Done()
37		io.Copy(w, r)
38	}
39
40	wg.Add(2)
41	go copyFunc(os.Stdout, cmd.Stdout)
42	go copyFunc(os.Stderr, cmd.Stderr)
43
44	cmd.Wait()
45	wg.Wait()
46
47	if cmd.ExitCode() != 0 {
48		panic(cmd.ExitCode())
49	}
50}
