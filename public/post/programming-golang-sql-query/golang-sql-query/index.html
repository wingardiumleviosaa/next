[Golang] 查詢 SQL DB 的幾種匯出方式

Publish on: 2022-04-19 Classify at: Programming/Golang
Words: 476 Read:≈ 1mins

簡易紀錄一下 sql query 回傳(印出)查詢結果的三種不同方法，三種方式回傳的格式都差不多，只是型態可能不一樣。本文使用的 database 是 Oracle。

連線 

 1package main
 2
 3import (
 4	"database/sql"
 5	"fmt"
 6
 7	_ "github.com/godror/godror"
 8)
 9
10const (
11	username = "ula"
12	password = "123456"
13	host     = "10.90.1.207"
14	port     = 1521
15	sid      = "emesHY"
16)
17
18func main() {
19
20	oralInfo := fmt.Sprintf("%s/%s@%s:%d/%s", username, password, host, port, sid)
21	db, err := sql.Open("godror", oralInfo)
22
23	err = db.Ping()
24	if err != nil {
25		fmt.Println("cant connect")
26		db.Close()
27	}
28
29	sqlComand := fmt.Sprintf("SELECT DISTINCT SERIAL_NUMBER,PRE_KPSN,THIS_KPSN,DATECODE,LOTCODE from EMESC.VP_ASSY_SUMMARY_SMT where serial_number = 'TBCBB2039913'")
30
31	rows, err := db.Query(sqlComand)
32	if err != nil {
33		fmt.Println("Query Database Failed!")
34	}
35	defer rows.Close()
36
37	// 插入以下其中一種作法
38}

作法一 

 1columns, err := rows.Columns()
 2if err != nil {
 3	fmt.Println(err)
 4}
 5count := len(columns)
 6tableData := make([]map[string]interface{}, 0)
 7values := make([]interface{}, count)
 8valuePtrs := make([]interface{}, count)
 9for rows.Next() {
10	for i := 0; i < count; i++ {
11		valuePtrs[i] = &values[i]
12	}
13	rows.Scan(valuePtrs...)
14	entry := make(map[string]interface{})
15	for i, col := range columns {
16		var v interface{}
17		val := values[i]
18		b, ok := val.([]byte)
19		if ok {
20			v = string(b)
21		} else {
22			v = val
23		}
24		entry[col] = v
25	}
26	tableData = append(tableData, entry)
27}
28jsonData, err := json.Marshal(tableData)
29if err != nil {
30	fmt.Println(err)
31}
32// fmt.Println(string(jsonData))
33
34var result []map[string]interface{}
35
36if err := json.Unmarshal([]byte(jsonData), &result); err != nil {
37	fmt.Println("Can't parse the json string")
38}
39
40fmt.Println(result)

作法二 

 1type Kpsn struct {
 2	SN   string `json:"Serial_Number"`
 3	Pre  string `json:"Pre_KPSN"`
 4	This string `json:"This_KPSN"`
 5	DC   string `json:"Datecode"`
 6	LC   string `json:"Lotcode"`
 7}
 8
 9var kpsns []*Kpsn
10
11for rows.Next() {
12	k := new(Kpsn)
13	rows.Scan(&k.SN, &k.Pre, &k.This, &k.DC, &k.LC)
14	kpsns = append(kpsns, k)
15}
16jsonStr, _ := json.Marshal(&kpsns)
17
18var result []map[string]interface{}
19// var result []Kpsn
20
21if err := json.Unmarshal([]byte(jsonStr), &result); err != nil {
22	fmt.Println("Can't parse the json string")
23}
24
25fmt.Println(result)

作法三 

 1var sn, pre, this, dc, lc string
 2var result []map[string]interface{}
 3
 4for rows.Next() {
 5	rows.Scan(&sn, &pre, &this, &dc, &lc)
 6	var tmp map[string]interface{}
 7	tmp = make(map[string]interface{})
 8	tmp["SERIAL_NUMBER"] = sn
 9	tmp["PRE_KPSN"] = pre
10	tmp["THIS_KPSN"] = this
11	tmp["DATECODE"] = dc
12	tmp["LOTCODE"] = lc
13	result = append(result, tmp)
14}
15fmt.Println(result)
