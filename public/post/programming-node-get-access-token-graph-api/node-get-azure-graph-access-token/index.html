[Node.js] Get Access Token 並使用 Microsoft Graph API 操作 AD Application

Publish on: 2022-10-23 Classify at: Programming/Node.js
Words: 283 Read:≈ 1mins

情景

使用 NodeJS @azure/msal-node module 取得 Access Token，並進一步使用 Microsoft Graph API 對操作 Azure AD Application 做 Role Assignment。

作法 

 1import * as https from 'https';
 2import fetch from 'node-fetch';
 3import * as msal from '@azure/msal-node';
 4
 5const httpsAgent = new https.Agent({
 6    rejectUnauthorized: false,
 7    keepAlive: true
 8});
 9
10const msalConfig = {
11    auth: {
12        clientId: '<YOUR CLIENT ID>',
13        clientSecret: '<YOUR CLIENT SECRET>',
14        authority: 'https://login.microsoftonline.com/<YOUR TENANT ID>/'
15    },
16};
17
18async function getToken(): Promise<any> {
19    const cca = new msal.ConfidentialClientApplication(msalConfig);
20    try{
21        return await cca.acquireTokenByClientCredential({
22            scopes: ['https://graph.microsoft.com/.default'],
23        });
24    }catch(e){
25        console.log(`Get Azure Access Token Error: ${e}`);
26    }
27}
28
29async function addAzureOpsgenieAppRole(userPrincipalName: string): Promise<any> {
30    const authResponse = await getToken();
31    const response = await fetch(`https://graph.microsoft.com/v1.0/users/${userPrincipalName}`, {
32        method: 'get',
33        headers: {'Authorization': `Bearer ${authResponse.accessToken}`},
34        agent: httpsAgent
35    }).then( response => {
36        if(response.ok){
37            return response.json();
38        }else{
39            return Promise.reject(response);
40        }
41    }).then( data => {
42        const createBody = {
43            principalId: data.id, 
44            resourceId: '<YOUR RESOURCE ID>',
45            appRoleId: '<YOUR APPLICATION ROLE ID>'
46        }
47        return fetch(`https://graph.microsoft.com/v1.0/users/${data.id}/appRoleAssignments`, {
48            method: 'post',
49            body: JSON.stringify(createBody),
50            headers: {'Content-Type': 'application/json', 'Authorization': `Bearer ${authResponse.accessToken}`},
51            agent: httpsAgent
52        }).then( response => {
53            if (response.ok) {
54                console.log('Successfully add user as assgined role to Azure AD Application')
55                return response.json();
56            } else {
57                return Promise.reject(response);
58            }
59        })
60    }).catch( e => {
61        console.warn('User Assignment already exists or error happens', e)
62    });
63}
